import java.io.*;
import java.util.*;

public class Main {

    static int n;
    static int[][] map;
    static int[][] dd = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    public static void main(String[] args) throws Exception {
        readData();
        int ans = solution();
        writeData(ans);
    }

    // ë¹„ê°€ ì ì  ë§ì´ ì˜¬ ë•Œ ì•ˆì „í•œ êµ¬ì—­ì˜ ìˆ˜ êµ¬í•˜ê¸°
    private static int solution() {
        int max = 0;
        for (int rain = 0; rain < 100; ++rain) {
            int safe = 0;
            boolean[][] visited = new boolean[n][n];
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (!visited[i][j] && map[i][j] > rain) {
                        rain(i, j, rain, visited);
                        safe += 1;
                    }
                }
            }
            if (safe == 0) return max;
            max = Math.max(max, safe);
        }
        return max;
    }

    // dfsë¡œ ì‹œì‘ì§€ì ê³¼ ì—°ê²°ëœ ì¹¨ìˆ˜ë˜ì§€ ì•Šì€ ì§€ì—­ íƒìƒ‰
    private static void rain(int r, int c, int prec, boolean[][] visited) {
        visited[r][c] = true;

        for (int dir = 0; dir < 4; ++dir) {
            int nr = r + dd[dir][0];
            int nc = c + dd[dir][1];

            if (nr < 0 || nr >= n || nc < 0 || nc >= n) continue; // ë²”ìœ„ë¥¼ ë²—ì–´ë‚œ ê²½ìš°
            if (visited[nr][nc] || map[nr][nc] <= prec) continue; // ì´ë¯¸ ë°©ë¬¸í–ˆê±°ë‚˜ ì¹¨ìˆ˜ëœ ì§€ì—­ì¸ ê²½ìš°
            rain(nr, nc, prec, visited);
        }
    }

    private static void readData() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
        map = new int[n][n];
        for (int i = 0; i < n; ++i) {
            map[i] = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
        }
        br.close();
    }

    private static void writeData(int ans) throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();
        sb.append(ans).append("\n");

        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

}
