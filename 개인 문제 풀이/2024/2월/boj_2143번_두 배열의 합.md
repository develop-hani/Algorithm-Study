import java.io.*;

    static long T; // ëª©í‘œ í•©

    private static List<Integer> getSums(int len, int[] nums) {

    private static long count(List<Integer> aList, List<Integer> bList) {

        int aIdx = 0;

        while (aIdx < aList.size() && bIdx >= 0) {
            int bValue = bList.get(bIdx);
