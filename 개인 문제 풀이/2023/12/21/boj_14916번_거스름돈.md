import java.io.*;
import java.util.Arrays;

public class Main {

    static int n;
    static int[] coins;

    // dpë¡œ ë™ì „ì˜ ê°œìˆ˜ë¥¼ ì„¸ëŠ” í•¨ìˆ˜
    private static void countCoins() {
        for (int i = 4; i <= n; i++) {
            if (i - 5 < 0 && coins[i - 2] > 0) {
                coins[i] = coins[i - 2] + 1;
                continue;
            }

            if (coins[i - 2] > 0 && coins[i - 5] > 0) coins[i] = Math.min(coins[i - 2] + 1, coins[i - 5] + 1);
            else if (coins[i - 2] > 0) coins[i] = coins[i - 2] + 1;
            else if (coins[i - 5] > 0) coins[i] = coins[i - 5] + 1;
        }
    }

    // ë°ì´í„° ì½ê¸° ë° ì´ˆê¸°í™”
    private static void readData() throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        n = Integer.parseInt(br.readLine());
        coins = new int[n + 1];
        Arrays.fill(coins, -1);

        if (n >= 2) coins[2] = 1;
        if (n >= 5) coins[5] = 1;

        br.close();
    }

    // ë°ì´í„° ì“°ê¸°
    private static void writeData() throws Exception {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();

        sb.append(coins[n]).append("\n");

        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

    public static void main(String[] args) throws Exception {
        readData();
        countCoins();
        writeData();
    }
}
