import java.io.*;
import java.util.*;

public class Main {

    static int N;
    static PriorityQueue<Node> pq = new PriorityQueue<>();
    static int[] parent;

    static class Node implements Comparable<Node> {
        int from, to;
        double cost;

        public Node(int from, int to, double cost) {
            this.from = from;
            this.to = to;
            this.cost = cost;
        }

        public int compareTo(Node node) {
            return (this.cost - node.cost) > 0 ? 1 : -1;
        }
    }

    // ì„œë¡œì†Œ ì§‘í•© ì´ˆê¸°í™”
    private static void init() {
        parent = new int[N];
        for (int i = 0; i < N; ++i) {
            parent[i] = i;
        }
    }

    // xì˜ ë¶€ëª¨ë¥¼ ì°¾ëŠ” í•¨ìˆ˜
    private static int findSet(int x) {
        if (x == parent[x]) return x;
        return parent[x] = findSet(parent[x]);
    }

    // xì™€ yë¥¼ unioní•˜ëŠ” í•¨ìˆ˜
    private static boolean union(int x, int y) {
        int rootX = findSet(x);
        int rootY = findSet(y);

        if (rootX == rootY) return false;
        parent[rootX] = parent[rootY];
        return true;
    }

    private static double kruskal() {
        init();
        double res = 0.0;
        int v = 0; // ì—°ê²°ëœ ê°„ì„ ì˜ ìˆ˜

        while (v < N - 1) {
            Node cur = pq.poll();
            if (union(cur.from, cur.to)) { // ì„œë¡œì†Œ ì§‘í•©ì´ì˜€ë‹¤ë©´ ì—°ê²°í•˜ê¸°
                res += cur.cost;
                v += 1;
            }
        }
        return res;
    }

    public static void main(String[] args) throws IOException {
        readData();
        double ans = kruskal();
        writeData(ans);
    }

    // ë‘ ë³„ ì‚¬ì´ì˜ ê±°ë¦¬
    private static double getDist(double x1, double y1, double x2, double y2) {
        double distX = Math.pow((x1 - x2), 2);
        double distY = Math.pow((y1 - y2), 2);
        return Math.sqrt(distX + distY);
    }

    private static void readData() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());

        double[][] stars = new double[N][2];
        for (int i = 0; i < N; ++i) {
            stars[i] = Arrays.stream(br.readLine().split(" ")).mapToDouble(Double::parseDouble).toArray();
            double x = stars[i][0];
            double y = stars[i][1];
            for (int j = 0; j < i; ++j) {
                pq.offer(new Node(i, j, getDist(x, y, stars[j][0], stars[j][1])));
            }
        }

        br.close();
    }

    private static void writeData(double ans) throws IOException {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();
        sb.append(String.format("%.2f\n", ans));
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

}
