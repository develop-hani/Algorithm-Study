import java.io.*;

    private static int[] countAlphabets(String str) {
        int[] cnt = new int[26];
        for (int i = 0; i < str.length(); ++i) {
            ++cnt[str.charAt(i) - 'a'];
        }
        return cnt;

    private static int[] getLen(int[] cnt, String str, int k) {

        int max = -1;

        for (int i = 0; i < str.length(); ++i) {
            char ch = str.charAt(i);

    }

    public static void main(String[] args) throws Exception {
