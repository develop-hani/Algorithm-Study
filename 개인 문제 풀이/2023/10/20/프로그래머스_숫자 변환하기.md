import java.util.*;

class Solution {
    public int solution(int x, int y, int n) {
        int[] count = new int[y + 1];
        Arrays.fill(count, -1);
        
        Queue<Integer> q = new LinkedList<>();
        q.offer(x);
        count[x] = 0;
        
        while(!q.isEmpty()) {
            int cur = q.poll();
            
            if (cur + n <= y && count[cur + n] < 0) { // cur + nìœ¼ë¡œ ì´ë™
                q.offer(cur + n);
                count[cur + n] = count[cur] + 1;
            }
            
            if (cur * 2 <= y && count[cur * 2] < 0) { // cur * 2ìœ¼ë¡œ ì´ë™
                q.offer(cur * 2);
                count[cur * 2] = count[cur] + 1;
            }
            
            if (cur * 3 <= y && count[cur * 3] < 0) { // cur * 3ìœ¼ë¡œ ì´ë™
                q.offer(cur * 3);
                count[cur * 3] = count[cur] + 1;
            }
            
        }
        
        
        return count[y];
    }
}
