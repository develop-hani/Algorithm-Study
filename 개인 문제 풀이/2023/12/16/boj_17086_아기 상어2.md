import java.io.*;
import java.util.*;

public class Main {

    static int n, m;
    static int[][] map;

    static Queue<Shark> q = new LinkedList<>();
    static boolean[][] visited;

    // ìœ„ìª½ë¶€í„° ì‹œê³„ë°©í–¥
    static int[] dr = {-1, -1, 0, 1, 1, 1, 0, -1};
    static int[] dc = {0, 1, 1, 1, 0, -1, -1, -1};

    static class Shark {
        int r, c, dist;
        public Shark (int r, int c, int dist) {
            this.r = r;
            this.c = c;
            this.dist = dist;
        }
    }

    // ë³€ìˆ˜ ì…ë ¥ ë°›ê¸°
    private static void readData() throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());

        map = new int[n][m];
        visited = new boolean[n][m];

        for (int i = 0; i < n; ++i) {
            st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < m; ++j) {
                map[i][j] = Integer.parseInt(st.nextToken());
                if (map[i][j] == 1) { // ìƒì–´ê°€ ìˆëŠ” ì¹¸ì´ë©´
                    q.offer(new Shark(i, j, 0));
                    visited[i][j] = true;
                }
            }
        }

        br.close();
    }

    // ìƒì–´ê°€ ìˆëŠ” ì¹¸ë¶€í„° bfs
    private static int getDistance() {
        int dist = -1;
        while (!q.isEmpty()) {
            Shark cur = q.poll();
            dist = cur.dist;

            for (int dir = 0; dir < 8; ++dir) {
                int nr = cur.r + dr[dir];
                int nc = cur.c + dc[dir];

                if (nr < 0 || nr >= n || nc < 0 || nc >= m) continue; // ë²”ìœ„ë¥¼ ë²—ì–´ë‚œ ê²½ìš°
                if (visited[nr][nc]) continue; // ì´ë¯¸ ë°©ë¬¸í•œ ê²½ìš°

                q.offer(new Shark(nr, nc, cur.dist + 1));
                visited[nr][nc] = true;
            }
        }

        return dist;
    }

    // ë°ì´í„° ì“°ê¸°
    private static void writeData(int dist) throws Exception {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();

        sb.append(dist).append("\n");

        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }

    public static void main(String[] args) throws Exception {
        readData();
        int max = getDistance();
        writeData(max);
    }
}
