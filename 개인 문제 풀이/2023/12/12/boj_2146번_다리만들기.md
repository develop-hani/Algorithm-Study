import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main {

    static int n;
    static int[][] map;

    static int[] dr = {-1, 1, 0, 0};
    static int[] dc = {0, 0, -1, 1};

    static Queue<Island> border = new LinkedList<>(); // ì„¬ì˜ ê°€ì¥ìë¦¬ ì €ì¥

    static class Island {
        int r, c; // ì„¬ì˜ ìœ„ì¹˜

        public Island(int r, int c) {
            this.r = r;
            this.c = c;
        }
    }

        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();

        // ì…ë ¥
        n = Integer.parseInt(br.readLine());

        map = new int[n][n];
        for (int i = 0; i < n; ++i) {
            map[i] = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();

        bw.write(sb.toString());

    // mapì˜ ì˜ë¯¸ (0:ë°”ë‹¤, 1:ëŒ€ë¥™, 2ì´ìƒ:ì„¬ì˜ ë²ˆí˜¸)
    static void findIslands() {
        int islandNum = 2;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {

                    numberIsland(i, j, islandNum++);

    // bfsë¡œ ì£¼ë³€ ì„¬ì„ íƒìƒ‰í•˜ë©° ì„¬ì— ë²ˆí˜¸ ë¶€ì—¬í•˜ê¸°
    static void numberIsland(int r, int c, int num) {
        Queue<Island> island = new LinkedList<>();
        boolean[][] visited = new boolean[n][n]; // ê°€ì¥ìë¦¬ ë°©ë¬¸ ì—¬ë¶€ ì €ì¥
        map[r][c] = num;
        island.offer(new Island(r, c));

        while (!island.isEmpty()) {
            Island cur = island.poll();

            for (int dir = 0; dir < 4; ++dir) {
                int nr = cur.r + dr[dir];
                int nc = cur.c + dc[dir];

                if (!validatePos(nr, nc)) continue; // ì§€ë„ì˜ ë²”ìœ„ë¥¼ ë²—ì–´ë‚œ ê²½ìš°

                if (map[nr][nc] != 1) continue; // ë°”ë‹¤ì´ê±°ë‚˜ ì´ë¯¸ ë°©ë¬¸í•œ ê²½ìš°

    // ë‘ ì„¬ì„ ì—°ê²°í•˜ëŠ” ë‹¤ë¦¬(ê°„ì²™ì§€) ê¸¸ì´ì˜ ìµœì†Ÿê°’ êµ¬í•˜ê¸°

            }
        }

        return min;
    }

    static boolean validatePos(int r, int c) {
        return r >= 0 && r < n && c >= 0 && c < n;
    }
}
